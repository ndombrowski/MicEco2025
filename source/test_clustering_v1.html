Minimal R example:

library(tidyverse)

# Function to convert FASTQ quality line (ASCII) to numeric Phred scores
fastq_to_phred <- function(qual_line, offset = 33) {
    # as.integer() converts the raw hex value to decimal
    # offset = 33 for Sanger/Illumina 1.8+ format
    as.integer(charToRaw(qual_line)) - offset
}

# Calculate Arithmetic Mean
arith_mean <- function(q) mean(q)

# Calculate Phred-aware Mean
# arithmetic averaging of Q scores is inappropriate because Q is logarithmic
# here: transform Q to linear probabilities before calculating the mean
phred_mean <- function(q) {
    error_prop <- 10^(-q / 10)
    avg_error <- mean(error_prop)
    avg_phred <- -10 * log10(avg_error)
    return(avg_phred)
}

# Generate quality lines for three random reads representing Illumina/Nanopore data
qual_lines <- c("IGD", "BCA", "?0&")
read_names <- paste0("read", seq_along(qual_lines))
reads <- setNames(lapply(qual_lines, fastq_to_phred), read_names)

# Df with all results 
df <- map_dfr(names(reads), function(i) {
    q <- reads[[i]]

    data.frame(
        read = i,
        raw_read = paste(q, collapse = ","),
        arithmetic = arith_mean(q),
        phred_aware = phred_mean(q)
    )
})

print(df, row.names = F)

Summary thoughts

NGSpeciesID merges clusters based on --aligned_threshold

With nanopore errors (~1–2%), raising –aligned_threshold to >0.995 likely won’t help, because the sequencing errors mask the small differences between closely related taxa.
